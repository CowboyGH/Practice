Задание 2. Заполните массив последовательными нечетными числами, начиная с 1;

namespace MyWorkSpace;

public class Program
{
    static void Main(String[] args)
    {
        const int N = 100;
        int num = 2;
        int[] nums = new int[N];

        for (int i = 0; i < N; i++)
        {
            nums[i] = 1 + i * num;
        }
    }
}
